jQuery

jQuery——write less do more

jQuery:优秀的js框架

jQuery的使用

  1. 引入jquery文件
    <script type="text/javascript" src="/js/jquery.min.js"></script>

  2. 页面加载事件

    //页面加载完之后再执行代码，先于window.onload执行
$(function(){

})
  3. jquery中$和jQuery
    $ === jQuery
    如果其他框架里面也使用了$符号，就使用jQuery的写法

  4. DOM对象和jquery对象的转换

    $(function(){
    //dom对象
    var domObj = document.getElementById("mySpan");

    //jquery对象
    var jqueryObj = $("#mySpan");

    //有时候需要把dom对象转成jquery对象--可以调用jquery里面的方法
    var domToJquery = $(domObj);

    //有时候需要把jquery对象转成dom对象--可以使用dom原生的属性和方法
    var jqueryToDom = jqueryObj[0];
})

  5. 选择器

    //id选择器 id一般不重复
$("#id");

//class选择器
$(".class");

//标签选择器 页面上的所有标签都有效果
$("div");

//获取所有元素--没什么用
$("*");

//多元素选择器
$("div,span")

//层次选择器
//祖先后代
$("div span")
//父亲儿子
$("div > span")
//相邻选择器
$("input + input")
//同级选择器
$("form ~ input")

//属性选择器

  6. jquery注册事件方式
    //方式一注册事件
    $(“#id”).click(function(){
    })

    //方式二注册事件
    $(“#id”).on(‘click’,function(){
    })
    //取消事件
    $(“#id”).off(‘click’)

    //方式三注册事件
    $(“#id”).bind(‘click’,function(){
    })
    //取消事件
    $(“#id”).unbind(‘click’);
